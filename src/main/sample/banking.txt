shared int money = 0;
function int addfive(int a){
    return a+5;
}
parallel {
thread {   int wait = 100; while (wait > 0){
        wait = wait - 1;
        lock
           money = money + 1;
        unlock
    }
}
thread {
    int wait = 100; while (wait > 0){
        wait = wait - 1;
        lock
            money = money - 1;
        unlock
    }
}
thread {
    lock
         money = addfive(money);;
    unlock
}
}
